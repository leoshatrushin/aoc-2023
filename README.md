f - file \
p - pixel map \

r - row \
c - column \
h,w - height, width \
i, j - row, column indices \
a - character, general purpose variable \
b - bool, general purpose variable \
d - direction variable \
e - general purpose variable \
g,z - functions \
n - list of numbers or number variable \
m - map, memo \
k,v - key, value \
l - list \
u - subtotal \
t - total
